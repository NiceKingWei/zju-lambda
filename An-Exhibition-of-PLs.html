Prototypical Inheritance

Use module and table to perform OOP

var EventModule = {               // singleton/object in scala
  'New':function(title) {
      var instance = {};
      instance.title = title;
      instance.audiences = [];
      return instance;
  },
  'join':function(instance,name){
    instance.audiences.push(name);
  }
}
var e = EventModule.New("zju-lambda speech");
EventModule.join(e,"nicekingwei");

How about the inheritance?

// just copy and modify the module
var SpecialEventModule = {};
for(key in EventModule) 
  SpecialEventModule[key] = EventModule[key];

  var instance = EventModule.New("special");
  instance.theme = theme;
  return instance;
}
SpecialEventModule.spec_fun = function(instance) {
  console.log("special");
}
var e = SpecialEventModule.New("monad");
SpecialEventModule.spec_fun(e);

Ghost type finds its place

Continuation

cont in continuation.js (by ByVoid)

function textProcessing(ret) {
  fs.readFile('somefile.txt', 'utf-8', cont(err, contents));
  if (err) return ret(err);
  contents = contents.toUpperCase();
  fs.readFile('somefile2.txt', 'utf-8', cont(err, contents2));
  if (err) return ret(err);
  contents += contents2;
  fs.writeFile('somefile_concat_uppercase.txt', contents, cont(err));
  if (err) return ret(err);
  ret(null, contents);
}
textProcessing(cont(err, contents));
if (err)
  console.error(err);

auto-run generator

function run(fn) {
  var gen = fn();

  function next(err, data) {
    var result = gen.next(data);
    if (result.done) return;
    result.value(next);
  }

  next();
}
run(gen);

Measure type

[<measure>] type g
let weight = 8.<g>
[<measure>] type ml
let vol = 100.f<ml>
let x = weight/vol

Optional Type

let x:Student? = fun()
let y = x!
let z = x?.name
if let w = x {
  // do something
}

Dependent Type

vadd : Vect n Int -> Vect n Int -> Vect n Int
vadd []        []        = []
vadd (x :: xs) (y :: ys) = x + y :: vadd xs ys

Refinement Type

{-@ zero :: Zero @-} 
zero = 0 :: Int 

{-@ one, two, three :: NonZero @-}
one = 1 :: Int
two = 2 :: Int

{-@ type Nat   = {v:Int | 0 <= v}        @-}
{-@ type Even  = {v:Int | v mod 2 == 0 } @-}
{-@ type Lt100 = {v:Int | v < 100}       @-}
